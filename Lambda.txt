import json
import boto3

dynamodb = boto3.resource('dynamodb')
table = dynamodb.Table('RegistroEventos')

def lambda_handler(event, context):
    detail = json.loads(event['detail'])

    table.put_item(
        Item={
            'id': detail['estado'],
            'temperatura': str(detail['temp']),
            'status': 'CRITICO'
        }
    )

    return {"status": "ok"}
